#include <Arduino.h>
#include <Bounce2.h>
#define pinBotao1 5
#define pinBotao2 19
#define pinBotao3 21

// * instancias
Bounce Botao1 = Bounce();
Bounce Botao2 = Bounce();
Bounce Botao3 = Bounce();

void setup()
{
  Botao1.attach(pinBotao1, INPUT_PULLUP);
  Botao2.attach(pinBotao2, INPUT_PULLUP);
  Botao3.attach(pinBotao3, INPUT_PULLUP);
  Serial.begin(9600);
}

void loop()
{
  Botao1.update();
  Botao2.update();
  Botao3.update();

  static int contagem = 0;

  static bool modoSelecao = 0;

  static int intensidadeA = 0;
  static int intensidadeB = 0;
  static int intensidadeC = 0;
  static int intensidadeD = 0;

  if (Botao1.fell())
  {
    if (modoSelecao == 1)
    {
      contagem++;
      if (contagem == 4)
        contagem = 0;
      {
        Serial.println(contagem);
      }
    }
    else
    {
      if (contagem == 0)

      {
        if (intensidadeB < 100)
          intensidadeB = intensidadeB + 10;
      }
      else
      {
        if (intensidadeA < 100)
          intensidadeA = intensidadeA + 10;
          Serial.println(intensidadeA);
      }
    }
  }

  if (Botao2.fell())
  {

    if (contagem == 3)
      contagem = contagem;
    {
      if (contagem == 0)
        contagem = 4;
      {
        contagem--;
        Serial.println(contagem);
      }
    }
  }

  if (Botao3.fell())
  {
    modoSelecao = !modoSelecao;
    Serial.println("botao3 presionado");
  }
}
